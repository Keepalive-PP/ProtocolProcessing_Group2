84 def blitteds(dst, srcs, copy = True):
85  """Blit multiple images on top of dst resizing it if needed"""
86  left = top = right = bottom = 0
87  dw, dh = dst.get_size()
88 
89  # Calculate the needed extra size
90  for src, (x, y) in srcs:
91  sw, sh = src.get_size()
92  left = max(left, -x)
93  top = max(top, -y)
94  right = max(right, x + sw - dw)
95  bottom = max(bottom, y + sh - dh)
96 
97  # Create a new surface if the old size is not enough
98  if top or left or bottom or right:
99  tmp = pygame.Surface((dw + left + right, dh + top + bottom))
100  tmp.set_colorkey(dst.get_colorkey())
101  tmp.fill(dst.get_colorkey())
102  tmp.blit(dst, (left, top))
103  dst = tmp
104  elif copy:
105  dst = dst.copy()
106 
107  # Blit the images to the appropriate positions
108  for src, (x, y) in srcs:
109  dst.blit(src, (x + left, y + top))
110 
111  return dst, (left, top)

159  rect = tileimage.get_rect()
160  if rect.width % width or rect.height % height:
161  print 'Tile image should be divisible to (%d,%d)' % (width, height)

164  #print "parse_tiles - got %dx%d tileset with %d cols and %d rows" % (rect.width, rect.height, cols, rows)
165  images = []
166  for y in range(rows):

183 def load_map(name):
184  path = os.path.join(MAPDIR, name)
185  with file(path, 'rb') as f:
186  aliases = {}
187  width = height = 0
188  spawns = {}
189  for line in f:
190  cmd = line.strip().split()
191  if not cmd:
192  break
193  if cmd[0] == 'SIZE':
194  width, height = int(cmd[1]), int(cmd[2])
195  elif cmd[0] == 'ALIAS':
196  aliases[cmd[1]] = cmd[2]
197  elif cmd[0] == 'SPAWN':
198  spawns.setdefault(int(cmd[1]), []).append((int(cmd[2]), int(cmd[3])))
199  else:
200  raise ValueError, 'Unknown map directive'
201  result = []
202  for y in range(height):
203  line = f.next()
204  tiles = [aliases.get(tile, tile) for tile in line.split()]
205  assert len(tiles) == width
206  result.append(tiles)
207  return result, (width, height), spawns
